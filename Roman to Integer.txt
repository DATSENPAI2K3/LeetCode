var romanToInt = function(s) {
    let roman = new Map([
        ["I", 1],["V", 5],["X", 10],["L", 50],["C", 100],["D", 500],["M", 1000]
    ]);
    let x = 0;
    for(let i = 0; i < s.length; i++){
        if(roman.get(s[i]) < roman.get(s[i + 1])){
            x += roman.get(s[i + 1]) - roman.get(s[i]);
            i++;
        }else x += roman.get(s[i]);
    }
    return x;
};
